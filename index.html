#!/bin/bash

LOVE=~/works
COUNT=$(find $LOVE -type f | wc -l)

if [ $COUNT -lt 3 ]; then
    HEART=$(echo "< 3" | sed -e 's/ //g')
    echo "$HEART"
fi

3. Run script.
$ sudo chmod a+x love.sh↵
$ ./love.sh↵
